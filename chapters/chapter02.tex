    如果需要撰写算法流程，参考算法\ref{alg:bits-back_decode}
\begin{algorithm}[htbp]
  \caption{Partial Bits-Back Decode}
  \label{alg:bits-back_decode}
  \KwIn{bitstream $m$}
  \KwOut{Gaussians[1:N]}
  
  $(m, N - K) \leftarrow$ ans-decode($m$) \tcp*{decode number of remaining Gaussians}
  
  $(m, G[K+1:N]) \leftarrow$ ans-decode($m$) \tcp*{decode $G[K+1:N]$}
  
  $(m, G[K+1:N]) \leftarrow$ ans-encode($m$, $\mathcal{U}_{(N-K)!}$, $G[K+1:N]$) \tcp*{re-encode with uniform distribution}
  
  $(m, G[1:K]) \leftarrow$ ans-decode($m$) \tcp*{decode $G[1:K]$}
  
  \KwRet{$G[1:N]$}
\end{algorithm}
